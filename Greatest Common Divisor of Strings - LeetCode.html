class Solution(object): def gcdOfStrings(self, str1, str2): """ :type str1: str :type str2: str :rtype: str """ def gcd(a, b): while b: a, b = b, a % b return a # Check if str1 + str2 equals str2 + str1 if str1 + str2 != str2 + str1: return "" # Calculate the gcd of the lengths of str1 and str2 gcd_length = gcd(len(str1), len(str2)) # The largest string x that divides both str1 and str2 return str1[:gcd_length] # Example Usage solution = Solution() print(solution.gcdOfStrings("ABCABC", "ABC")) # Output: "ABC" print(solution.gcdOfStrings("ABABAB", "ABAB")) # Output: "AB" print(solution.gcdOfStrings("LEET", "CODE")) # Output: "
